8 object JoiningWithOutcomeExample extends IOApp {
9   override def run(args: List[String]): IO[ExitCode] = {
10     for {
11       fiberA <- (IO
12         .println("Hello I am Fiber A")
13         .foreverM
14         .timeoutTo(3.seconds, IO.unit)
15         >> IO.pure(1)).start
16       fiberB <- (IO
17         .println("Hello I am Fiber B")
18         .foreverM
19         .timeoutTo(3.seconds, IO.unit)
20         >> IO.pure(1)).start
21 
22       /* Attendo il completamento */
23       a <- fiberA.joinWithNever
24       b <- fiberB.join flatMap {
25         case Outcome.Succeeded(fb) => fb
26         case Outcome.Errored(e)    => MonadError[IO, Throwable].raiseError(e)
27         /* Nel caso in cui il fiber figlio venga interrotto, si cerca di
28         interrompere l'attuale fiber, se non è possibile si cade in deadlock */
29         case Outcome.Canceled() => MonadCancel[IO].canceled >> Spawn[IO].never
30       }
31       _ <- IO.println(s"a: $a, b: $b")
32     } yield ExitCode.Success
33   }
