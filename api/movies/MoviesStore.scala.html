69       jsonDecoder: EntityDecoder[F, Json] = jsonOf[F, Json]): F[Double] = {
70     val informationMovieUrl = IMDB.getIMDBMovieInfoUrl(title.replaceAll(" ", "%20"))

82     } yield rating
