33       /**
34        * Update the director on a specific movie
35        */
36       case req @ PUT -> Root / "api" / "director" :? MovieIdQueryParamMatcher(movieId) =>
37         moviesStore.getMovieById(UUID.fromString(movieId)).flatMap {
38           case Some(_) =>
39             req

41               .flatMap(newDirector => moviesStore.updateDirectorInAMovie(movieId, newDirector))
42               .flatMap(_ => Ok("Updated successfully"))
43           case _ => NotFound(s"Movie with id $movieId not found")
44         }
45     }
