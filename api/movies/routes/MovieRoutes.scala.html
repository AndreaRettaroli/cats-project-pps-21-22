27   def routes[F[_]: Async](moviesStore: MoviesStore[F]): HttpRoutes[F] = {
28     val client: Client[F] = JavaNetClientBuilder[F].create
29     val dsl = new Http4sDsl[F] {}
30     import dsl._
31 
32     HttpRoutes.of[F] {
33       case GET -> Root =>
34         Ok("Hello World!")
35 
36       /**
37        * Get movie by ID
38        */
39       case GET -> Root / "api" / "movie" / UUIDVar(movieId) =>
40         moviesStore
41           .getMovieById(movieId)
42           .flatMap(maybeMovie =>
43             maybeMovie.fold(NotFound(s"Wrong id $movieId"))(movie => Ok(movie.asJson))
44           )
45 
46       /**
47        * Create movie
48        */
49       case req @ POST -> Root / "api" / "movie" =>
50         for {
51           movie <- req.decodeJson[Movie]
52           addedMovie <- moviesStore.createMovie(movie)
53           response <- Created(addedMovie.asJson)
54         } yield response
55 
56       /**
57        * Update Movie
58        */
59       case req @ PUT -> Root / "api" / "movie" / UUIDVar(movieId) =>
60         for {
61           maybeMovie <- moviesStore.getMovieById(movieId)
62           movieToAdd <- req.decodeJson[Movie]
63           response <- maybeMovie.fold(NotFound(s"No movie with id $movieId found"))(movie =>
64             moviesStore.updateMovie(movieId, movieToAdd) >> Ok(
65               new MovieWithId(movieId.toString, movieToAdd).asJson
66             )
67           )
68         } yield response
69 
70       /**
71        * Delete movie
72        */
73       case DELETE -> Root / "api" / "movie" / UUIDVar(movieId) =>
74         for {
75           maybeMovie <- moviesStore.getMovieById(movieId)
76           response <- maybeMovie.fold(NotFound(s"No movie with id $movieId found"))(movie =>
77             moviesStore.deleteMovie(movieId) >> Ok(s"Successfully deleted movie ${movie.id}")
78           )
79         } yield response
80 
81       /**
82        * Get movies list directed by director, can also filter them per year
83        */
84       case GET -> Root / "api" / "movies" :? DirectorQueryParamMatcher(
85             director
86           ) +& OptionalYearQueryParamMatcher(maybeYear) =>
87         moviesStore
88           .getFilteredMoviesByDirector(director)
89           .flatMap(movies =>
90             if (movies.isEmpty) NotFound(s"No results found for: $director")
91             else
92               maybeYear.fold(Ok(movies.asJson)) {
93                 case Invalid(_) => BadRequest("Invalid year passed")
94                 case Valid(year) =>
95                   val moviesInYear = movies.filter(_.movie.year === year.getValue)
96                   if (moviesInYear.nonEmpty) Ok(moviesInYear.asJson)
97                   else NotFound(s"No movies directed for ${director} in ${year}")
98               }
99           )
100 
101       /**
102        * Get movies list by year
103        */
104       case GET -> Root / "api" / "movies" :? YearQueryParamMatcher(maybeYear) =>
105         def findMoviesBy(year: Int): F[Response[F]] = for {
106           movies <- moviesStore.getFilteredMoviesByYear(year)
107           response <-
108             if (movies.nonEmpty) Ok(movies.asJson) else NotFound(s"No movies in ${year} found")
109         } yield response
110 
111         maybeYear.fold(_ => BadRequest("Invalid year passed"), year => findMoviesBy(year.getValue))
112 
113       /**
114        * Get movies list by genre
115        */
116       case GET -> Root / "api" / "movies" :? GenreQueryParamMatcher(genre) =>
117         moviesStore
118           .getFilteredMoviesByGenre(genre)
119           .flatMap(movies =>
120             if (movies.isEmpty) NotFound(s"No movie found with genre: $genre")
121             else Ok(movies.asJson)
122           )
123 
124       /**
125        * Get movies list by actor
126        */
127       case GET -> Root / "api" / "movies" :? ActorQueryParamMatcher(actor) =>
128         moviesStore
129           .getFilteredMoviesByActor(actor)
130           .flatMap(movies =>
131             if (movies.isEmpty) NotFound(s"No movie found with actor: $actor")
132             else Ok(movies.asJson)
133           )
134 
135       /**
136        * Get movies list
137        */
138       case GET -> Root / "api" / "movies" =>
139         moviesStore.getAllMovies.flatMap(movies =>
140           if (movies.nonEmpty) Ok(movies.asJson) else NoContent()
141         )
142 
143       /**
144        * Get movies rating
145        */
146       case GET -> Root / "api" / "movies" / "rating" =>
147         for {
148           movies <- moviesStore.getAllMovies
149           titles = movies.map(_.movie.title)
150           movieRatingPairsList <- titles.parTraverse(title =>
151             moviesStore.getMoviesRating(title, client).map(r => (title, r))
152           )
153           (bestTitle, score) = movieRatingPairsList.maxBy(_._2)
154           response <- Ok(
155             s"${movieRatingPairsList.map(t => s"${t._1}: ${t._2}").mkString("; ")}. The best movie is $bestTitle with a score of $score"
156           )
157         } yield response
158 
159     }
160   }
