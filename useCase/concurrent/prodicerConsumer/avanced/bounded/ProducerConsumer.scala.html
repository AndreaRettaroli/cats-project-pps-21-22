17   def consumer[F[_]: Async: Console](id: Int, state: Ref[F, State[F, Int]]): F[Unit] = {
18     val consume: F[Int] =
19       Deferred[F, Int].flatMap { taker =>
20         state.modify {
21           /* If queue is not empty and there aren't producers blocked */
22           case State(capacity, queue, takers, offerers) if queue.nonEmpty && offerers.isEmpty =>
23             val (i, rest) = queue.dequeue
24             State(capacity, rest, takers, offerers) -> Async[F].pure(i)
25           /* If queue is not empty and there are producers blocked */
26           case State(capacity, queue, takers, offerers) if queue.nonEmpty =>
27             val (i, rest) = queue.dequeue
28             val ((_, release), tail) = offerers.dequeue
29             State(capacity, rest, takers, tail) -> release.complete(()).as(i)
30           /* If there are producers blocked */
31           case State(capacity, queue, takers, offerers) if offerers.nonEmpty =>
32             val ((i, release), rest) = offerers.dequeue
33             State(capacity, queue, takers, rest) -> release.complete(()).as(i)
34           /* Otherwise */
35           case State(capacity, queue, takers, offerers) =>
36             State(capacity, queue, takers.enqueue(taker), offerers) -> taker.get
37         }.flatten
38       }
39     for {
40       i <- consume
41       _ <- Console[F].println(s"Consumer $id has got item: $i")
42       _ <- consumer(id, state)
43     } yield ()
44   }

46   def producer[F[_]: Async: Console](id: Int, counter: Ref[F, Int], state: Ref[F, State[F, Int]]): F[Unit] = {
47     def produce(i: Int): F[Unit] =
48       Deferred[F, Unit].flatMap { offerer =>
49         state.modify {
50           /* If there are takers blocked */
51           case State(capacity, queue, takers, offerers) if takers.nonEmpty =>
52             val (taker, rest) = takers.dequeue
53             State(capacity, queue, rest, offerers) -> taker.complete(i).void
54           /* If queue size < capacity */
55           case State(capacity, queue, takers, offerers) if queue.size < capacity =>
56             State(capacity, queue.enqueue(i), takers, offerers) -> Async[F].unit
57           /* Otherwise */
58           case State(capacity, queue, takers, offerers) =>
59             State(capacity, queue, takers, offerers.enqueue(i -> offerer)) -> offerer.get
60         }.flatten
61       }
62 
63     for {
64       i <- counter.getAndUpdate(_ + 1) // Update the item
65       _ <- produce(i)
66       _ <- Console[F].println(s"Producer $id product item: $i")
67       _ <- producer(id, counter, state)
68     } yield ()
69   }
