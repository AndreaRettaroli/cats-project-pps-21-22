22   def consumer[F[_]: Async: Console](id: Int, state: Ref[F, State[F, Int]]): F[Unit] = {
23     val consume: F[Int] =
24       Deferred[F, Int].flatMap { taker =>
25         Async[F].uncancelable { poll =>
26           state.modify {
27             /* If queue is not empty and there aren't producers blocked */
28             case State(capacity, queue, takers, offerers) if queue.nonEmpty && offerers.isEmpty =>
29               val (i, rest) = queue.dequeue
30               State(capacity, rest, takers, offerers) -> Async[F].pure(i)
31             /* If queue is not empty and there are producers blocked */
32             case State(capacity, queue, takers, offerers) if queue.nonEmpty =>
33               val (i, rest) = queue.dequeue
34               val ((_, release), tail) = offerers.dequeue
35               State(capacity, rest, takers, tail) -> release.complete(()).as(i)
36             /* If there are producers blocked */
37             case State(capacity, queue, takers, offerers) if offerers.nonEmpty =>
38               val ((i, release), rest) = offerers.dequeue
39               State(capacity, queue, takers, rest) -> release.complete(()).as(i)
40             /* Otherwise */
41             case State(capacity, queue, takers, offerers) =>
42               /* Cleanup taker by takerers queue */
43               val cleanup = state.update { s => s.copy(takers = s.takers.filter(_ ne taker))}
44               State(capacity, queue, takers.enqueue(taker), offerers) -> poll(taker.get).onCancel(cleanup)
45           }.flatten
46         }
47       }
48     for {
49       i <- consume
50       _ <- Console[F].println(s"Consumer $id has got item: $i")
51       _ <- consumer(id, state)
52     } yield ()
53   }

55   def producer[F[_]: Async: Console](id: Int, counter: Ref[F, Int], state: Ref[F, State[F, Int]]): F[Unit] = {
56     def produce(i: Int): F[Unit] =
57       Deferred[F, Unit].flatMap { offerer =>
58         Async[F].uncancelable { poll =>
59           state.modify {
60             /* If there are takers blocked */
61             case State(capacity, queue, takers, offerers) if takers.nonEmpty =>
62               val (taker, rest) = takers.dequeue
63               State(capacity, queue, rest, offerers) -> taker.complete(i).void
64             /* If queue size < capacity */
65             case State(capacity, queue, takers, offerers) if queue.size < capacity =>
66               State(capacity, queue.enqueue(i), takers, offerers) -> Async[F].unit
67             /* Otherwise */
68             case State(capacity, queue, takers, offerers) =>
69               /* Cleanup offerer by offerers queue */
70               val cleanup = state.update { s => s.copy(offerers = s.offerers.filter(_._2 ne offerer))}
71               State(capacity, queue, takers, offerers.enqueue(i -> offerer)) -> poll(offerer.get).onCancel(cleanup)
72           }.flatten
73         }
74       }
75 
76     for {
77       i <- counter.getAndUpdate(_ + 1) // Update the item
78       _ <- produce(i)
79       _ <- Console[F].println(s"Producer $id product item: $i")
80       _ <- producer(id, counter, state)
81     } yield ()
82   }
