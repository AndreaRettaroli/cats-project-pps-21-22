9 object ProducerConsumer {
10   case class State[F[_], A](queue: Queue[A], takers: Queue[Deferred[F, A]])
11 
12   object State {
13     def empty[F[_], A]: State[F, A] = State(Queue.empty, Queue.empty)
14   }
15 
16   def consumer[F[_] : Async : Console](id: Int, state: Ref[F, State[F, Int]]): F[Unit] = {
17     val consume: F[Int] = {
18       Deferred[F, Int].flatMap {
19         taker =>
20           state.modify {
21             case State(queue, takers) if queue.nonEmpty => // If queue is not empty
22               val (i, rest) = queue.dequeue
23               State(rest, takers) -> Async[F].pure(i) // Got element in queue, we can just return it
24             case State(queue, takers) => State(queue, takers.enqueue(taker)) -> taker.get // No element in queue, must block caller until some is available
25           }.flatten
26       }
27     }
28 
29     for {
30       i <- consume
31       _ <- Console[F].println(s"Consumer $id has got item: $i")
32       _ <- consumer(id, state)
33     } yield ()
34   }
35 
36   def producer[F[_] : Sync : Console](id: Int, counter: Ref[F, Int], state: Ref[F, State[F, Int]]): F[Unit] = {
37     def produce(i: Int): F[Unit] =
38       state.modify {
39         case State(queue, takers) if takers.nonEmpty => // If there is almost a taker dequeue it and release it
40           val (taker, rest) = takers.dequeue
41           State(queue, rest) -> taker.complete(i).void
42         case State(queue, takers) => // If there aren't takers produce element
43           State(queue.enqueue(i), takers) -> Sync[F].unit
44       }.flatten
45 
46     for {
47       i <- counter.getAndUpdate(_ + 1) // Update the item
48       _ <- produce(i)
49       _ <- Console[F].println(s"Producer $id product item: $i")
50       _ <- producer(id, counter, state)
51     } yield ()
52   }
