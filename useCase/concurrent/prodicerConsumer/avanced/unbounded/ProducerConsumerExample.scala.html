10  * producer and consumer with unbounded buffer.
11  * To terminate the execution use CTRL-C || CTRL+F2
12  */
13 
14 object ProducerConsumerExample extends IOApp {
15   override def run(args: List[String]): IO[ExitCode] = {
16     for {
17       state <- Ref.of[IO, State[IO, Int]](State.empty[IO, Int])
18       counter <- Ref.of[IO, Int](1)
19       producers = List.range(1, 11).map(producer(_, counter, state)) //-> 10 producers
20       consumers = List.range(1, 11).map(consumer(_, state)) //-> 10 consumers
21       res <- (producers ++ consumers)
22         .parSequence.as(ExitCode.Success) // Run producer and consumer in parallel until done (cancelling with CTRL-C)
23         .handleErrorWith {
24           t => Console[IO].errorln(s"Error caught: ${t.getMessage}").as(ExitCode.Error)
25         }
26     } yield res
27   }
