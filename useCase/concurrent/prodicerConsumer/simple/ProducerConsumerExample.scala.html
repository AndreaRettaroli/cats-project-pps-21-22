10 /**
11  * This solution is a basic implementation of concurrency problem of
12  * producer and consumer. This solution is inefficient because producers runs faster than the consumer
13  * so the queue is constantly growing.
14  *
15  * To terminate the execution use CTRL-C
16  */
17 
18 object ProducerConsumerExample extends IOApp {
19   override def run(args: List[String]): IO[ExitCode] = {
20     for {
21       queue <- Ref.of[IO, Queue[Int]](Queue.empty[Int])
22       res <- (consumer(queue), producer(queue, 0))
23         .parMapN((_, _) => ExitCode.Success) // Run producer and consumer in parallel until done
24         .handleErrorWith { t =>
25           Console[IO].errorln(s"Error caught: ${t.getMessage}").as(ExitCode.Error)
26         }
27     } yield res
28   }
29 }
