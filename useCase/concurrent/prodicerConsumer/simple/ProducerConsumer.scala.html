15   def producer[F[_]: Sync: Console](queue: Ref[F, Queue[Int]], counter: Int): F[Unit] = {
16     for {
17       _ <- if(counter % 10000 == 0) Console[F].println(s"Produced item $counter ") else Sync[F].unit

21   }

23   def consumer[F[_]: Sync: Console](queue: Ref[F, Queue[Int]]): F[Unit] = {
24     for {
25       iO <- queue.modify { q =>
26         q.dequeueOption.fold((q, Option.empty[Int])) { // remove element to the queue
27           case (i, q) => (q, Option(i))
28         }
29       }
30       _ <- if (iO.nonEmpty) Console[F].println(s"Consumed item: ${iO.get}") else Sync[F].unit
31       _ <- consumer(queue)
32     } yield ()
33   }
