35   def inputOutputStreams[F[_] : Sync](inputFile: File, outputFile: File): Resource[F, (InputStream, OutputStream)] = {
36     for {

42   private def transmit[F[_] : Sync](originFile: InputStream, destinationFile: OutputStream, buffer: Array[Byte],
43                                     acc: Long): F[Long] = {
44     for {
45       amount <- Sync[F].blocking(originFile.read(buffer, 0, buffer.length))
46       count <- if (amount > -1) Sync[F].blocking(destinationFile.write(buffer, 0, amount)) >> transmit(originFile,
47         destinationFile, buffer, acc + amount) else Sync[F].pure(acc)
48     } yield count
49   }

51   private def transfer[F[_] : Sync](originFile: InputStream, destinationFile: OutputStream, bufferSize: Int): F[Long] =
52     transmit(originFile, destinationFile, new Array[Byte](bufferSize), 0L)

54   /**
55    *  Method that copy data from file to another.
56    *
57    * @param originFile  file of data to copy
58    * @param destinationFile file where the data need to be copied
59    * @param bufferSize optional buffer size, by default is BUFFER_SIZE
60    * @tparam F the effect type
61    * @return returns an IO instance with the byte copied.
62    */
63 
64   def copy[F[_] : Sync](originFile: File, destinationFile: File, bufferSize: Option[Int] = None): F[Long] = {
65     inputOutputStreams(originFile, destinationFile).use { case (in, out) =>
66       transfer(in, out, bufferSize.getOrElse(BUFFER_SIZE))
67     }
